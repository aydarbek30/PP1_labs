#include <iostream>
#include <string>


using namespace std;


int main(){
    int size;
    cin >> size;
    int arr[size];
    for(int i = 0; i < size; i++){
        cin >> arr[i];
    }
    int n;
    if(n < size){
        cout << arr[n] * arr[n];
        return 0;
    }
    cout << -1;
    return 0;
}
